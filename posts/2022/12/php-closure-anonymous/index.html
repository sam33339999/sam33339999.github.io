class TestAnonymous
{
    public function __construct(
    	protected string $name
	) { /** PHP 8 Syntax, With Construct initial attributes */ }

    public function hello(): void
    {
        echo "Hello world - {$this->name} \n";
    }

    public function helloWithAnonymous()
    {
        $newClass = new Class('Sunny') extends TestAnonymous
        {
            public function hello(): void
            {
                parent::hello();
                echo "Hello John - {$this->name} \n";
            }
        }; // 這裡記得要有分號。
		$newClass->hello();
    }
    
}

$ta = new TestAnonymous('Sam');
$ta->hello();
$ta->helloWithAnonymous();
