theme = "diary" # 這裡的 theme 會去查詢你目錄下的 themes 資料夾下面，是否有無相同名稱的資料夾，來當作佈景主題

[markup]
  [markup.highlight]
    codeFences = true   # 程式碼圍欄
    guessSyntax = true  # 猜語法
    hl_Lines = ""
    lineNoStart = 1     # 行數從 1 起算
    lineNos = true      # 是否顯示行號
    lineNumbersInTable = true # 表格內是否顯示行號
    # https://gohugo.io/content-management/syntax-highlighting/
    noClasses = true    
    #style = "perldoc"  # 原本的樣式，但是在深色模式下看不清楚
    style = "monokai"   # 程式碼區塊使用的 style
    tabWidth = 4        # 預設縮排為多少空白
